function debounce(fn, delay){
    let timer = null;
    return function(...args){
        if(timer)clearTimeout(timer);
        timer = setTimeout(()=>{
            fn.apply(this, args);
        }, delay);
    }
}

    return function(...args){
        const now = new Date.getTime();
        if((now-last)>=delay){
            fn.apply(this, args);
            last = now;

# 6. 深拷贝

```js
function deepcopy(obj) {
  // 处理 null/undefined
  if (obj === null || typeof obj !== 'object') return obj;

  // 处理 Date
  if (obj instanceof Date) return new Date(obj);
  
  // 处理正则
  if (obj instanceof RegExp) return new RegExp(obj);
  
  // 处理 Error
  if (obj instanceof Error) {
    const err = new Error(obj.message);
    err.stack = obj.stack; // 保留堆栈信息更完整
    return err;
  }

  // 处理函数（深拷贝函数最简单可靠的方式）
  if (typeof obj === 'function') return obj.bind({});

  // 处理数组 / 对象
  const newObject = Array.isArray(obj) ? [] : {};

  // 遍历自身可枚举属性
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      newObject[key] = deepcopy(obj[key]);
    }
  }

  return newObject;
}
```

# 7. 数组去重

**适用场景**：最常见的情况，不需要考虑对象。
#### 方案 A：利用 Set

#### （⭐️ 最推荐，一行搞定）

```javascript
function uniqueBasic(arr) {
  // 展开运算符(...)或者 Array.from 都可以
  return [...new Set(arr)]; 
}
// 原理：Set 内部使用 SameValueZero 算法判断重复，连 NaN 也能去重
```
#### 方案 B：利用 filter + indexOf

#### （⭐️ 面试常考，考逻辑）

```javascript
function uniqueBasic(arr) {
  return arr.filter((item, index) => {
    // 如果当前元素在数组中第一次出现的索引 === 当前索引，说明是首次出现
    return arr.indexOf(item) === index;
  });
}
// 缺点：时间复杂度 O(n²)，每次 indexOf 都会遍历一次数组
```
#### 方案 C：利用 Map 的键名不能重复

#### （⭐️ 考察对数据结构的灵活运用）

```javascript
function uniqueBasic(arr) {
  const map = new Map();
  return arr.filter(item => !map.has(item) && map.set(item, true));
}
// 优点：时间复杂度 O(n)，比 filter+indexOf 性能好很多
```

### 💡 面试加分话术（总结陈词）
> “关于数组去重，我不会一上来就写代码，而是先问清楚数据场景：
> 1. 如果只是基本类型，直接 `[...new Set(arr)]`，时间复杂度 O(n)，最优雅；
> 2. 如果是后端返回的列表对象，通常会根据 `id` 去重，我会用 `Map` 配合 `filter`，性能最好；
> 3. 如果要求深度去重（长得一样就去掉），我**不会用 `JSON.stringify`**，因为它有键顺序的坑且无法处理特殊类型，我会手写一个排序键名的深度序列化函数配合 `Map` 来做。”

# 12. 对象比较

```js
function isEqual(a, b) {
    // 1. 严格相等直接返回 true（包含基础类型、引用同一对象）
    if (a === b) return true;

    // 2. 处理 NaN：NaN === NaN 为 false，所以单独判断
    if (Number.isNaN(a) && Number.isNaN(b)) return true;

    // 3. 排除 null：typeof null === 'object'，必须提前过滤
    if (a === null || b === null) return false;

    // 4. 只有两个都是对象时，才进行递归深比较
    if (typeof a === 'object' && typeof b === 'object') {
        const keysA = Object.keys(a);
        const keysB = Object.keys(b);

        // 键数量不同直接不相等
        if (keysA.length !== keysB.length) return false;

        // 递归比较每个键的值
        return keysA.every(key => isEqual(a[key], b[key]));
    }

    // 5. 其他情况全部不相等
    return false;
}
```

# 13. 数组转树

function toTree(list, parId) {
    // 递归函数
    function loop(pid) {
        const res = [];
        // 直接遍历 list，不依赖外部 len，更规范
        for (const item of list) {
            if (item.pid === pid) {

                if (children.length) {
                    item.children = children;
                }
                res.push(item);
            }
        }
        return res;
    }

    return loop(parId);
}

# 16. 深拷贝

```js
function deepCopy(obj, cache = new WeakMap()) {
    // 1. 处理基本数据类型 + 函数
    if (obj === null || typeof obj !== "object") {
        return obj;
    }

    // 2. 处理循环引用（核心修复）
    if (cache.has(obj)) {
        return cache.get(obj);
    }

    // 3. 处理内置对象
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof RegExp) return new RegExp(obj);
    if (obj instanceof Error) return new Error(obj.message);
    if (obj instanceof Map) {
        const mapCopy = new Map();
        cache.set(obj, mapCopy);
        obj.forEach((val, key) => {
            mapCopy.set(key, deepCopy(val, cache));
        });
        return mapCopy;
    }
    if (obj instanceof Set) {
        const setCopy = new Set();
        cache.set(obj, setCopy);
        obj.forEach(val => {
            setCopy.add(deepCopy(val, cache));
        });
        return setCopy;
    }

    // 4. 创建新对象/数组
    const newObj = Array.isArray(obj) ? [] : {};
    // 缓存当前对象，解决循环引用
    cache.set(obj, newObj);

    // 5. 遍历所有键：包含 普通键 + Symbol键
    Reflect.ownKeys(obj).forEach(key => {
        newObj[key] = deepCopy(obj[key], cache);
    });

    return newObj;
}
```

# 24. 异步并发数限制

```js
// 核心：控制最多同时跑 count 个异步任务
async function limit(count, array, fn) {
  const result = []
  const doing = [] // 正在执行的任务

  for (const item of array) {
    // 启动任务
    const p = fn(item).then(res => result.push(res))
    doing.push(p)

    // 超过并发数，就等一个完成再继续
    if (doing.length >= count) {
      await Promise.race(doing)
      // 清理已完成的任务（不清理也能跑，清理更干净）
      doing.splice(doing.indexOf(p), 1)
    }
  }

  // 等所有剩下的任务跑完
  await Promise.all(doing)
  return result
}
```

async function parallelSum(...nums) {

// 极简大数相加
function bigAdd(a, b) {
  let i = a.length - 1
  let j = b.length - 1
  let carry = 0
  let res = []

  while (i >= 0 || j >= 0 || carry) {
    // 取当前位，没有就取 0
    const num1 = +a[i--] || 0
    const num2 = +b[j--] || 0

    // 相加
    const sum = num1 + num2 + carry
    // 存个位
    res.push(sum % 10)
    // 算进位
    carry = (sum / 10) | 0
  }

  // 反转得到结果
  return res.reverse().join('')
}

console.log(bigAdd('1', '2097')) // 2098
